## The Tool Selection Problem

**Scenario:** You give your agent 20 tools. The agent uses wrong ones constantly.

**Research shows:** Agent accuracy decreases with tool count:
- 1-5 tools: 92% correct selection
- 6-10 tools: 84% correct selection
- 11-20 tools: 71% correct selection
- 20+ tools: 58% correct selection

**Why:** LLMs pattern-match descriptions. Large option spaces overwhelm them.

## Advanced Optimization Techniques

### 1. Retrieval-Augmented Tool Selection

**Pattern:** Use retrieval to pre-filter tools before agent selection.

<CodeGroup>
```python Python
from sentence_transformers import SentenceTransformer

class ToolRetriever:
    """Retrieve relevant tools for query."""
    
    def __init__(self, all_tools: list[dict]):
        self.tools = all_tools
        self.embedder = SentenceTransformer('all-MiniLM-L6-v2')
        
        # Pre-compute embeddings
        self.tool_embeddings = self.embedder.encode([
            f"{t['name']}: {t['description']}" 
            for t in all_tools
        ])
    
    def retrieve_relevant_tools(
        self,
        query: str,
        top_k: int = 5
    ) -> list[dict]:
        """Get most relevant tools for query."""
        
        # Embed query
        query_emb = self.embedder.encode([query])[0]
        
        # Calculate similarity
        from sklearn.metrics.pairwise import cosine_similarity
        similarities = cosine_similarity(
            [query_emb],
            self.tool_embeddings
        )[0]
        
        # Get top-k
        top_indices = similarities.argsort()[-top_k:][::-1]
        
        return [self.tools[i] for i in top_indices]

# Usage
retriever = ToolRetriever(all_100_tools)

async def agent_with_retrieval(query: str):
    # Stage 1: Retrieve relevant subset (fast)
    relevant_tools = retriever.retrieve_relevant_tools(query, top_k=5)
    
    # Stage 2: Agent selects from small set (accurate)

        model="claude-sonnet-4-20250514",
        messages=[{"role": "user", "content": query}],
        tools=relevant_tools  # Only 5 tools vs 100
    )
    
    # Result: 92% accuracy (vs 58% with all 100 tools)
```

```ts TypeScript
import { SentenceTransformer } from "@xenova/transformers";

class ToolRetriever {
    private tools: any[];
    private embedder: any;
    private toolEmbeddings: any;

    /** Retrieve relevant tools for query. */
    constructor(allTools: any[]) {
        this.tools = allTools;
        this.embedder = await SentenceTransformer.from_pretrained('all-MiniLM-L6-v2');
        
        // Pre-compute embeddings
        this.toolEmbeddings = await this.embedder.encode(
            allTools.map(t => `${t.name}: ${t.description}`)
        );
    }

    async retrieveRelevantTools(
        query: string,
        topK: number = 5
    ): Promise<any[]> {
        // Get most relevant tools for query
        
        // Embed query
        const queryEmb = await this.embedder.encode([query]);
        
        // Calculate similarity (cosine similarity)
        const similarities = this.toolEmbeddings.map((emb: any, i: number) => {
            const dotProduct = queryEmb[0].reduce((sum: number, val: number, idx: number) => sum + val * emb[idx], 0);
            const normA = Math.sqrt(queryEmb[0].reduce((sum: number, val: number) => sum + val * val, 0));
            const normB = Math.sqrt(emb.reduce((sum: number, val: number) => sum + val * val, 0));
            return { index: i, similarity: dotProduct / (normA * normB) };
        });
        
        // Get top-k
        const topIndices = similarities
            .sort((a, b) => b.similarity - a.similarity)
            .slice(0, topK)
            .map(item => item.index);
        
        return topIndices.map(i => this.tools[i]);
    }
}

// Usage
const retriever = new ToolRetriever(all100Tools);

async function agentWithRetrieval(query: string) {
    // Stage 1: Retrieve relevant subset (fast)
    const relevantTools = await retriever.retrieveRelevantTools(query, 5);
    
    // Stage 2: Agent selects from small set (accurate)
    const response = await client.messages.create({
        model: "claude-sonnet-4-20250514",
        messages: [{ role: "user", content: query }],
        tools: relevantTools  // Only 5 tools vs 100
    });
    
    // Result: 92% accuracy (vs 58% with all 100 tools)
}
```

### 4. Tool Usage Analytics

**Pattern:** Track and optimize based on actual usage.

<CodeGroup>
```python Python
class ToolAnalytics:
    """Monitor tool usage patterns."""
    
    def __init__(self):
        self.stats = defaultdict(lambda: {
            "calls": 0,
            "successes": 0,
            "failures": 0,
            "avg_latency_ms": 0
        })
    
    async def call_with_analytics(
        self,
        tool_name: str,
        params: dict
    ):
        """Track metrics for every call."""
        
        start = time.time()
        
        try:
            result = await execute_tool(tool_name, params)
            
            # Update stats
            stats = self.stats[tool_name]
            stats["calls"] += 1
            stats["successes"] += 1
            
            latency = (time.time() - start) * 1000
            stats["avg_latency_ms"] = (
                (stats["avg_latency_ms"] * (stats["calls"] - 1) + latency) 
                / stats["calls"]
            )
            
            return result
            
        except Exception as e:
            self.stats[tool_name]["failures"] += 1
            raise
    
    def get_recommendations(self) -> list[str]:
        """Analyze and suggest optimizations."""
        
        recommendations = []
        
        for tool, stats in self.stats.items():
            # Unused tools
            if stats["calls"] == 0:
                recommendations.append(
                    f"Remove unused tool: {tool}"
                )
            
            # High failure rate
            failure_rate = stats["failures"] / max(stats["calls"], 1)
            if failure_rate > 0.3:
                recommendations.append(
                    f"Tool '{tool}' fails {failure_rate:.0%} of the time. "
                    f"Review error handling or simplify parameters."
                )
            
            # High latency
            if stats["avg_latency_ms"] > 2000:
                recommendations.append(
                    f"Tool '{tool}' averages {stats['avg_latency_ms']:.0f}ms. "
                    f"Consider caching or optimization."
                )
        
        return recommendations
```

```ts TypeScript
class ToolAnalytics {
    /** Monitor tool usage patterns. */
    private stats: Record<string, {
        calls: number;
        successes: number;
        failures: number;
        avg_latency_ms: number;
    }> = {};

    constructor() {
        // Initialize stats
    }

    async callWithAnalytics(
        toolName: string,
        params: any
    ): Promise<any> {
        // Track metrics for every call
        const start = Date.now();
        
        try {
            const result = await executeTool(toolName, params);
            
            // Update stats
            if (!this.stats[toolName]) {
                this.stats[toolName] = {
                    calls: 0,
                    successes: 0,
                    failures: 0,
                    avg_latency_ms: 0
                };
            }
            
            const stats = this.stats[toolName];
            stats.calls++;
            stats.successes++;
            
            const latency = Date.now() - start;
            stats.avg_latency_ms = (
                (stats.avg_latency_ms * (stats.calls - 1) + latency) 
                / stats.calls
            );
            
            return result;
        } catch (e) {
            if (!this.stats[toolName]) {
                this.stats[toolName] = {
                    calls: 0,
                    successes: 0,
                    failures: 0,
                    avg_latency_ms: 0
                };
            }
            this.stats[toolName].failures++;
            throw e;
        }
    }

    getRecommendations(): string[] {
        // Analyze and suggest optimizations
        const recommendations: string[] = [];
        
        for (const [tool, stats] of Object.entries(this.stats)) {
            // Unused tools
            if (stats.calls === 0) {
                recommendations.push(`Remove unused tool: ${tool}`);
            }
            
            // High failure rate
            const failureRate = stats.failures / Math.max(stats.calls, 1);
            if (failureRate > 0.3) {
                recommendations.push(
                    `Tool '${tool}' fails ${(failureRate * 100).toFixed(0)}% of the time. ` +
                    `Review error handling or simplify parameters.`
                );
            }
            
            // High latency
            if (stats.avg_latency_ms > 2000) {
                recommendations.push(
                    `Tool '${tool}' averages ${stats.avg_latency_ms.toFixed(0)}ms. ` +
                    `Consider caching or optimization.`
                );
            }
        }
        
        return recommendations;
    }
}
```

## Check Your Understanding

1. **Optimization:** You have 15 tools but only 5 are regularly used. What should you do?
   
   <details>
   <summary>Answer</summary>
   
   **Options:**
   
   1. **Remove unused tools** (if truly unnecessary)
   2. **Consolidate related tools** (merge similar functionality)
   3. **Use tool retrieval** (dynamically select relevant 5)
   4. **Phase-based filtering** (show different tools at different stages)
   
   Best approach: Combination of 1 and 2.
   - Remove tools with 0 usage
   - Consolidate tools with similar purpose
   - Result: Smaller, clearer tool set with higher accuracy
   </details>
